Que 2. Write code to explain prototype chaining

Ans:  function Person() {
        this.name = 'Nisha'
    }
    
    // adding property 
    Person.prototype.name = 'Peter';
    Person.prototype.age = 22
    
    const person1 = new Person();
    
    console.log(person1.name); 
    console.log(person1.age);
   output: Nisha 
           22
